library(ggplot2)

# Load your data
df <- read.csv("C:/Users/vtt89/OneDrive/Tissue_specificity/tau_lectin_family.csv", header = TRUE)
colnames(df) <- c("GeneID", "Tissue", "Tau", "Family")

# Convert tissue order if needed
df$Tissue <- factor(df$Tissue, levels = unique(df$Tissue))

# Plot
ggplot(df, aes(x = Tissue, y = Tau, fill = Tissue)) +
  geom_violin(trim = FALSE, alpha = 0.6) +
  geom_boxplot(width = 0.1, outlier.shape = NA, color = "black") +
  geom_hline(yintercept = 0.8, linetype = "dotted", color = "grey40", linewidth = 1) +
  facet_wrap(~ Family, scales = "free_y") +
  theme_minimal(base_size = 14) +
  labs(title = "Tissue-Specific Tau Index by Gene Family",
       x = "Tissue", y = "Tau Index") +
  theme(
    panel.grid = element_blank(),                      # Remove all internal grid lines
    panel.border = element_rect(colour = "black", fill = NA, linewidth = 1),  # Add black border around panels
    axis.line = element_line(colour = "black", linewidth = 1),                # Black axis lines
    axis.ticks = element_line(colour = "black", linewidth = 1),               # Black ticks
    axis.ticks.length = unit(0.25, "cm"),                                     # Tick length
    axis.text.x = element_text(angle = 45, hjust = 1),
    legend.position = "none"
  )
