library(pheatmap)
library("writexl")
library("readxl")

data=as.data.frame(read_excel("path to file"))
foldchange=data[,-1:-2]

annotation_row=data.frame(Group=data$"Type")
rownames(annotation_row)=data[,1]
rownames(foldchange)
rownames(foldchange)=data[,1]
pdf("path to file",width=25,height=30)
